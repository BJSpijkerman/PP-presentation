\begin{frame}{Wave Equation For Propagation}
	\vspace{-0.25cm}
	\begin{equation*}
		\begin{split}
			\nabla^2 \phi(\mathbf{r}) = \frac{\epsilon(\mathbf{r})}{c^2} \frac{\partial^2}{\partial t^2}\phi(\mathbf{r})\\
		\end{split}
	\end{equation*}
	\vspace{0.25cm}
	\begin{equation*}
		\phi(\mathbf{r}) = A e^{ i \left( \mathbf{k}\mathbf{r} - \omega t \right) } \hspace*{1cm} \epsilon(\mathbf{r}) =  \epsilon(x,y) \epsilon(z) = \epsilon(z)
	\end{equation*}
	\vspace{0.25cm}
	\begin{equation*}
		\begin{split}
			 \frac{\partial^2}{\partial z^2} \phi(z) - k_{x,y}^2\phi(z) = -\epsilon(z)\frac{\omega^2}{c^2} \phi(\mathbf{r})\\
		\end{split}
	\end{equation*}
\end{frame}
